document.getElementById("addAlbumBtn").addEventListener("click",function(){
    const albumname=document.getElementById("albumName").value;
    const releaseyear=document.getElementById("releaseYear").value;
    const genre=document.getElementById("genre").value;
    const checkbox=document.getElementById("favorite");

    if (!albumname || !genre || !releaseyear){
        document.getElementById("error-message").textContent="Please fill out all fields!";
        return;
    }
    document.getElementById("error-message").textContent="";

    let newele=document.createElement("li");
    const alblis=document.getElementById("albumList");
    const favlis=document.getElementById("favoriteAlbumList")
    newele.innerHTML=`${albumname} | Released: ${releaseyear} | Genre:${genre}`
    if (checkbox.checked){
        favlis.appendChild(newele);
    }else{
        alblis.appendChild(newele);
    }
    document.getElementById("albumName").value="";
    document.getElementById("releaseYear").value="";
    document.getElementById("genre").value="";
    document.getElementById("favorite").checked=false;
})
