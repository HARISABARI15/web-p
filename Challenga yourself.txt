document.getElementById("getGradeButton").addEventListener("click",function(){
    let val=document.getElementById("marks").value;
    const res=document.getElementById("gradeResult");
    if(!val){
        res.textContent="Enter any marks to get the grades.";
        return;
    }
    val=parseInt(val);
    if (val<0 || val>100){
        res.textContent="The marks should be between 0-100.";
        return;
    }
    if (val<60){
        res.textContent=`For marks ${val}, your grade is: F`;
    } else if(val<70){
        res.textContent=`For marks ${val}, your grade is: D`;
    } else if(val<80){
        res.textContent=`For marks ${val}, your grade is: C`;
    }else if(val<90){
        res.textContent=`For marks ${val}, your grade is: B`;
    }else{
        res.textContent=`For marks ${val}, your grade is: A`;
    }
})
