JS
const chckboxes=["chkWater","chkExercise","chkReading"];
document.querySelector("#mark-progress").addEventListener("click",function(){
    
    let count=0;
    for(let i =0;i<chckboxes.length;i++){
        const checkbox = document.getElementById(chckboxes[i]);
        if (checkbox.checked){
            count+=1;
        }
    }
    document.getElementById("summary").textContent="You have completed "+count+" habits today.";

    
})

document.querySelector("#reset-progress").addEventListener("click",function(){
    for(let i =0;i<chckboxes.length;i++){
        document.getElementById(chckboxes[i]).checked=false;
    }
    document.getElementById("summary").textContent="You have completed 0 habits today.";
    count=0;
})
